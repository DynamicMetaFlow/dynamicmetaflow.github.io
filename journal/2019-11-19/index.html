11:28 AM: Objects and Arrays #

The syntax for an object and an array in Javascript are similar to each other causing mistakes in syntax, especially when going from a Object -> Array.

Remember that in the array there is no need for a const, and that the brackets indicate the beginning and end of an element. Other oddities are the usage of the “,” and “[ ]“.

import React from 'react';
import ReactDOM from 'react-dom';

const App = () => {
  const course = {
    name: 'Half Stack application development',
    parts: [
    {
      name : 'Fundamentals of React:',
      exercises: 10
    },
    {
      name: 'Using props to pass data:',
      exercises:  7
    },
    {
      name: 'State of a component:',
      exercises: 14
    }
  ]
}
  return (
    <div>
      <Header course = {course.name} />
      <Content parts = {course.parts} />
      <Total parts = {course.parts} />
    </div>
  )
}

const Header = (props) => {
  return <h1>{props.course}</h1>
}

const Part = (props) => {
  return (
    <div>
      <p> {props.part.name} {props.part.exercises} </p>
    </div>
  )
}

const Content = ({parts}) => {
  return (
    <div>
      <Part part = {parts[0]} />
      <Part part = {parts[1]} />
      <Part part = {parts[2]} />
    </div>
  )
}

const Total = ({parts}) => {
  let total = parts[0].exercises + parts[1].exercises + parts[2].exercises;
  return (
    <div>
      <p> Number of exercises: {total} </p>
    </div>

  )
}

ReactDOM.render(<App />, document.getElementById('root'));
